45 REAL limit_lip (REAL (*f)(int,const REAL&),
46  int lip_value,
47  bool (*on_domain)(const REAL&),
48  const REAL& x)
49 {
50  if ( on_domain(x) != true ) iRRAM_REITERATE(0);
51 
52  limit_computation env;
53 
54  REAL x_new,lim;
55  sizetype x_error;
56 
57  x_new=x;
58  x_new.geterror(x_error);
59  assert(x_new.value);
60  sizetype_exact(x_new.error);
61 
62  limit_debug("starting limit_lip1");
63 
64  while (1) {
65  try{
66  iRRAM_DEBUG2(2,"trying to compute limit_lip1 with precision %d...\n",actual_stack().actual_prec);
67  lim=f(env.saved_prec(),x_new);
68  if (lim.error.exponent > env.saved_prec()) {
69  env.inc_step(2);
70  iRRAM_DEBUG2(2,"limit_lip1 too imprecise, increasing precision locally to %d...\n",actual_stack().actual_prec);
71  } else {
72  iRRAM_DEBUG2(2,"getting result with local error %d*2^(%d)\n",
73  lim.error.mantissa, lim.error.exponent);
74  break;
75  }}
76  catch ( Iteration it){
77  env.inc_step(2);
78  iRRAM_DEBUG2(2,"limit_lip1 failed, increasing precision locally to %d...\n",actual_stack().actual_prec);
79  } }
80  lim.adderror(sizetype_add_power2(x_error << lip_value, env.saved_prec()));
81  iRRAM_DEBUG2(2,"end of limit_lip1 with error %d*2^(%d)\n"
82  " error of argument: %d*2^(%d)\n",
83  lim.error.mantissa,lim.error.exponent,
84  x_error.mantissa,x_error.exponent);
85  return lim;
86 }

207 REAL lipschitz (REAL (*f)(const REAL&),
208  REAL (*lip_f)(const REAL&),
209  bool (*on_domain)(const REAL&),
210  const REAL& x)
211 {
212  if ( on_domain(x) != true ) iRRAM_REITERATE(0);
213 
214  REAL x_new,lip_result,lip_bound;
215  sizetype x_error;
216 
217  {
218  limit_computation env(0, stiff::rel{});
219  iRRAM_DEBUG1(2,"starting lipschitz1b ...\n");
220 
221 // for the computation of the Lipschitz bound, we work with
222 // reduced precision:
223  {
224  relaxed rel;
225  lip_bound=lip_f(x);
226  }
227  x_new=x;
228  x_new.geterror(x_error);
229  assert(x_new.value);
230  sizetype_exact(x_new.error);
231 
232 
233  bool try_it=true;
234  while (try_it) {
235  try { try_it=false;
236  lip_result=f(x_new); }
237  catch ( Iteration it) { try_it=true;
238  env.inc_step(2);
239  iRRAM_DEBUG2(2,"limit_lip2 failed, increasing precision locally to %d...\n",actual_stack().actual_prec);
240  }
241  }
242  }
243 
244  iRRAM_DEBUG2(2,"getting result with local error %d*2^(%d)\n",
245  lip_result.error.mantissa, lip_result.error.exponent);
246  lip_result.adderror((lip_bound.getsize() + lip_bound.geterror()) * x_error);
247  iRRAM_DEBUG2(2,"end of lipschitz_1b with error %d*2^(%d)\n"
248  " for argument with error %d*2^(%d)\n",
249  lip_result.error.mantissa,lip_result.error.exponent,
250  x_error.mantissa,x_error.exponent);
251  return lip_result;
252 }

390 REAL limit_hint (REAL (*f)(int, const REAL&),
391  int hint,
392  const REAL& x)
393 {
394  limit_computation env;
395 
396  REAL lim,limnew;
397  sizetype limnew_error;
398  sizetype lim_error;
399  int hintcopy=hint;
400  int success=0;
401 
402  int element=max(x.geterror().exponent,env.saved_prec());
403 
404  iRRAM_DEBUG1(2,"starting limit_hint1...\n");
405 
406  while (1) {
407  try {
408  iRRAM_DEBUG2(2,"trying to compute limit_hint1 with precicion 2^(%d)...\n",element);
409  limnew=f(element,x);
410  limnew_error = sizetype_add_power2(limnew.geterror(), element);
411  if ( (! success) || sizetype_less(limnew_error,lim_error) ) {
412  lim=limnew;
413  lim_error=limnew_error;
414  iRRAM_DEBUG2(2,"getting result with error %d*2^(%d)\n",
415  lim_error.mantissa, lim_error.exponent);
416  } else {
417  iRRAM_DEBUG1(2,"computation successful, but no improvement\n");
418  hintcopy=2*hintcopy;
419  }
420  success=1;
421  if (element>iRRAM_prec_array[1])break;
422  element=element+hint;
423  }
424  catch ( Iteration it) {
425  if (element>iRRAM_prec_array[1])break;
426  element=element+hintcopy;
427  }}
428  if ( ! success) {
429  iRRAM_DEBUG1(1,"computation of limit_hint1 failed\n");
430  iRRAM_REITERATE(0);
431  }
432  lim.seterror(lim_error);
433  iRRAM_DEBUG2(2,"end of limit_hint1 with error %d*2^(%d)\n",
434  lim_error.mantissa,lim_error.exponent);
435  return lim;
436 }

489  int lip,
490  bool (*on_domain)(const REALMATRIX&),
491  const REALMATRIX& x)
492 {
493  if ( on_domain(x) != true ) iRRAM_REITERATE(0);
494 
495  limit_computation env;
496 
497 
498  REALMATRIX x_new,lim,limnew;
499  sizetype lim_error;
500 
501  sizetype_exact(lim_error);
502  x_new=x; x_new.seterror(lim_error);
503 
504  limit_debug("starting limit_matrix_lip1");
505 
506  while (1) {
507  try {
508  iRRAM_DEBUG2(2,"trying to compute limit_matrix_lip1 with precision %d...\n",actual_stack().actual_prec);
509  limnew=f(env.saved_prec(),x_new);
510  lim=limnew;
511  lim.geterror(lim_error);
512  if (lim_error.exponent > env.saved_prec()) {
513  env.inc_step(2);
514  iRRAM_DEBUG2(2,"limit_lip too imprecise, increasing precision locally to %d...\n",actual_stack().actual_prec);
515  } else {
516  iRRAM_DEBUG2(2,"getting result with local error %d*2^(%d)\n",
517  lim_error.mantissa,lim_error.exponent);
518  break;
519  }}
520  catch ( Iteration it) {
521  env.inc_step(2);
522  iRRAM_DEBUG2(2,"limit_lip1 failed, increasing precision locally to %d...\n",actual_stack().actual_prec);
523  }}
524  sizetype_exact(lim_error);
525  lim.adderror(lim_error);
526  lim.adderror(x.geterror() << lip);
527  iRRAM_DEBUG2(2,"end of limit_matrix_lip1 with error %d*2^(%d)\n",
528  lim.geterror().mantissa,lim.geterror().exponent);

656 // general limit operator for FUNCTION objects on REAL numbers
657 //
658 // REAL limit ( FUNCTION<REAL,int> f )
659 // if FUNCTION f defines a normed Cauchy sequence, i.e. |f(i)-x|<= 2^{i}
660 // then limit(f) returns x
661 //********************************************************************************
662 
663 REAL limit (const FUNCTION<REAL,int> & f )
664 {
665  limit_computation env;
666 
667  REAL lim,limnew;
668  sizetype limnew_error;
669  sizetype lim_error;
670 
671  int element=env.saved_prec();
672  int element_step=env.saved_step();
673  int firsttime=2;
674 
675  limit_debug("starting limit_FUNCTION");
676 
677  while (1) {
678  try {
679  iRRAM_DEBUG2(2,"trying to compute limit_FUNCTION with precicion 2^(%d)...\n",element);
680  limnew=f(element);
681  limnew_error = sizetype_add_power2(limnew.geterror(), element);
682  if (firsttime ==2 ) if ( limnew_error.exponent > env.saved_prec(-1)) {
683  iRRAM_DEBUG2(2,"computation not precise enough (%d*2^%d), trying normal p-sequence\n",
684  limnew_error.mantissa, limnew_error.exponent);
685  element_step=1;
686  element=4+iRRAM_prec_array[element_step];
687  firsttime=1;
688  }
689  if ( firsttime != 0 || sizetype_less(limnew_error,lim_error) ) {
690  lim=limnew;
691  lim_error=limnew_error;
692  iRRAM_DEBUG2(2,"getting result with error %d*2^(%d)\n",
693  lim_error.mantissa, lim_error.exponent);
694  } else {
695  iRRAM_DEBUG1(2,"computation successful, but no improvement\n");
696  }
697  firsttime=0;
698  if (element<=env.saved_prec())break;
699  element_step+=4;
700  element=iRRAM_prec_array[element_step];
701  }
702  catch ( Iteration it) {
703  if ( firsttime==0) {
704  iRRAM_DEBUG1(2,"computation failed, using best success\n");
705  break;
706  } else
707  if ( firsttime==2) {
708  iRRAM_DEBUG1(2,"computation failed, trying normal p-sequence\n");
709  element_step=1;
710  element=4+iRRAM_prec_array[element_step];
711  firsttime=1;
712  } else {
713  iRRAM_DEBUG1(1,"computation of limit_FUNCTION failed totally\n");
714  iRRAM_REITERATE(0);
715  }}
716  }
717  lim.seterror(lim_error);
718  iRRAM_DEBUG2(2,"end of limit_FUNCTION with error %u*2^(%u)\n",
719  lim_error.mantissa, lim_error.exponent);
720  return lim;
721 }
