206 /* Requires for Result and ContArgs functions
207  * - int geterror_exp(const C &, const ContArgs &...)
208  * (usually it's enough to specialize the single-argument versions of
209  * geterror_exp(const C &) and geterror_exp(const ContArgs &)... or to
210  * overload geterror(const C &) and geterror(const ContArgs &)...;
211  * then the maximum-norm is used on the vector of all continuous parameters)
212  * - sizetype geterror(const Result &)
213  * - void seterror(Result &, const sizetype &)
214  */
223 template <typename F,typename... ContArgs>
224 auto limit(F f, const ContArgs &... cont_args)
225 -> typename std::enable_if<internal::any_continuous<ContArgs...>::value,
226  decltype(f(0,cont_args...))>::type
227 {
228  using Result = decltype(f(0,cont_args...));
229 
230  limit_computation env;
231 
232  Result lim, limnew;
233  sizetype limnew_error, lim_error;
234  int args_error_exp;
235 
236  int element = env.saved_prec();
237  int element_step = env.saved_step();
238  int firsttime = 2;
239 
240  args_error_exp = geterror_exp(cont_args...);
241 
242  limit_debug("starting general limit_gen1");
243 
244  while (1) {
245  try {
246  iRRAM_DEBUG2(2,"trying to compute general limit_gen1 "
247  "with precicion 2^(%d)...\n", element);
248  limnew = f(element,cont_args...);
249  limnew_error = sizetype_add_power2(geterror(limnew), element);
250  if (firsttime == 2)
251  if (limnew_error.exponent > env.saved_prec(-1)
252  && limnew_error.exponent > args_error_exp - env.saved_prec(-1)) {
253  iRRAM_DEBUG2(2,"computation not precise enough (%d*2^%d), "
254  "trying normal p-sequence\n",
255  limnew_error.mantissa,
256  limnew_error.exponent);
257  element_step = 1;
258  element = 4+iRRAM_prec_array[element_step];
259  firsttime = 1;
260  }
261  if (firsttime != 0 ||
262  sizetype_less(limnew_error,lim_error)) {
263  lim = limnew;
264  lim_error = limnew_error;
265  iRRAM_DEBUG2(2,"getting result with error %d*2^(%d)\n",
266  lim_error.mantissa, lim_error.exponent);
267  } else {
268  iRRAM_DEBUG1(2,"computation successful, but no improvement\n");
269  }
270  firsttime = 0;
271  if (element <= env.saved_prec())
272  break;
273  element_step += 4;
274  element = iRRAM_prec_array[element_step];
275  } catch (const Iteration &it) {
276  if (firsttime == 0) {
277  iRRAM_DEBUG1(2,"computation failed, using best success\n");
278  break;
279  } else if (firsttime == 2) {
280  iRRAM_DEBUG1(2,"computation failed, trying normal p-sequence\n");
281  element_step = 1;
282  element = 4+iRRAM_prec_array[element_step];
283  firsttime = 1;
284  } else {
285  iRRAM_DEBUG1(1,"computation of general limit_gen1 failed totally\n");
286  iRRAM_REITERATE(0);
287  }
288  }
289  }
290  seterror(lim, lim_error);
291  iRRAM_DEBUG2(2,"end of general limit_gen1 with error %d*2^(%d)\n",
292  lim_error.mantissa, lim_error.exponent);
293  return lim;
294 }

297 auto limit(F f, const DiscArgs &... disc_args)
298 -> typename std::enable_if<!internal::any_continuous<DiscArgs...>::value,
299  decltype(f(0,disc_args...))>::type

307 
308  limit_debug("starting general limit_0");

420 template <class ARGUMENT, class DISCRETE, class RESULT>
421 RESULT limit_lip (RESULT (*f)(int,const ARGUMENT&,DISCRETE param),
422  int lip_value,
423  bool (*on_domain)(const ARGUMENT&),
424  const ARGUMENT& x,DISCRETE param)
425 {
426  if ( on_domain(x) != true ) iRRAM_REITERATE(0);
427 
428  limit_computation env;
429 
430  ARGUMENT x_new;
431  RESULT lim;
432  sizetype lim_error,x_error;
433 
434  x_new=x;
435  x_new.geterror(x_error);
436  sizetype_exact(lim_error);
437  x_new.seterror(lim_error);
438 
439  limit_debug("starting limit_lip");
440 
441  while (1) {
442  try{
443  iRRAM_DEBUG2(2,"trying to compute limit_lip with precision %d...\n",actual_stack().actual_prec);
444  lim=f(env.saved_prec(),x_new,param);
445  lim.geterror(lim_error);
446  if (lim_error.exponent > env.saved_prec()) {
447  env.inc_step(2);
448  limit_debug2("limit_lip too imprecise");
449 
450  } else {
451  iRRAM_DEBUG2(2,"getting result with local error %d*2^(%d)\n",
452  lim_error.mantissa, lim_error.exponent);
453  break;
454  }}
455  catch ( Iteration it){
456  env.inc_step(2);
457  limit_debug2("limit_lip failed");
458  } }
459  lim_error = sizetype_power2(env.saved_prec());
460  lim.adderror(lim_error);
461  lim_error = x_error << lip_value;
462  lim.adderror(lim_error);
463  iRRAM_DEBUG2(2,"end of limit_lip with error %d*2^(%d)\n"
464  " error of argument: %d*2^(%d)\n",
465  lim.geterror().mantissa,lim.geterror().exponent,
466  x_error.mantissa,x_error.exponent);
467  return lim;
468 }

534 template <class CONT_ARGUMENT, class DISCRETE_ARGUMENT, class RESULT, class PARAM>
535 RESULT lipschitz_1p_1a (RESULT (*f)(const DISCRETE_ARGUMENT&, const PARAM& param),
536  int lip,
537  bool (*on_domain)(const CONT_ARGUMENT&, const PARAM& param),
538  const CONT_ARGUMENT& x, const PARAM& param )
539 {
540  if ( on_domain(x,param) != true ) iRRAM_REITERATE(0);
541 
542  limit_computation env(0);
543  iRRAM_DEBUG1(2,"starting lipschitz_1p_1a ...\n");
544 
545  DISCRETE_ARGUMENT x_center;
546  RESULT lip_result;
547  sizetype lip_error,x_error;
548 
549  bool try_it=true;
550  x.to_formal_ball(x_center,x_error);
551 
552  while (try_it) {
553  try { try_it=false;
554  lip_result=f(x_center,param); }
555  catch ( Iteration it) { try_it=true;
556  env.inc_step(2);
557  iRRAM_DEBUG2(2,"lipschitz_1p_1a failed, increasing precision locally to step %d...\n",actual_stack().prec_step);
558  }
559  }
560 
561  iRRAM_DEBUG2(2,"getting result with local error %d*2^(%d)\n",
562  lip_result.error.mantissa, lip_result.error.exponent);
563  lip_error = x_error << lip;
564  lip_result.adderror(lip_error);
565  iRRAM_DEBUG2(2,"end of lipschitz_1p_1a with error %d*2^(%d)\n"
566  " for argument with error %d*2^(%d)\n",
567  lip_result.geterror().mantissa,lip_result.geterror().exponent,
568  x_error.mantissa,x_error.exponent);
569  return lip_result;
