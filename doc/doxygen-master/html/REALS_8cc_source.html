88 sizetype REAL::double_pair::geterror() const
89 {
90  double rwidth = (upper_neg + lower_pos);
91  if (!std::isfinite(rwidth))
92  iRRAM_REITERATE(0);
93  int e;
94  unsigned m = (unsigned)ldexp(frexp(-rwidth, &e), 30) + 2;
95  // Here, the "+2" accounts for the possible truncation error and
96  // the error on the computation of cvalue.
97  /* round to -\infty => dp.upper-dp.lower <= -rwidth <= m*2^(e-30) */
98  // So we have that the interval (dp.lower,dp.upper) is a subset
99  // of the interval (cvalue - m*2^(e-29),cvalue + m*2^(e-29))
100  return sizetype_normalize({m, e - 29});
101 }

103 void REAL::mp_make_mp()
104 {
105  if (!std::isfinite(dp.upper_neg) || !std::isfinite(dp.lower_pos))
106  iRRAM_REITERATE(0);
107  if (dp.upper_neg == -dp.lower_pos) {
108  // here we have a point interval...
109  if (!value)
110  MP_init(value);
111  MP_double_to_mp(dp.lower_pos, value);
112  error = sizetype_exact();
113  } else {
114  // now we know that it is not a point interval:
115  MP_init(value);
116  MP_double_to_mp(dp.lower_pos, value);
117  MP_type value1;
118  MP_init(value1);
119  MP_double_to_mp(-dp.upper_neg, value1);
120  MP_type value2;
121  MP_init(value2);
122  MP_add(value, value1, value2, -1150);
123  MP_shift(value2, value, -1);
124  MP_clear(value1);
125  MP_clear(value2);
126  // value0 differs from precise result (lower+upper)/2 by at most
127  // 2^{-1200},
128  // which is sufficiently more precise than all bits even of the
129  // smallest double number (2^-1075), so here the error is as
130  // small as reasonable
131  error = dp.geterror();
132  }
133  MP_getsize(value, vsize);
134 }

136 void REAL::mp_from_mp(const REAL & y)
137 {
138  // d=MP_mp_to_double(y) is required to return a value d such that
139  // there is no other double between d and x.
140  // the conversion should also generate infinities in case of overflow
141  // we don't require a directed rounding here!
142  double center = MP_mp_to_double(y.value);
143  // in consequence, wd will be the negative of an upper bound
144  // for the width of the interval!
145  double wd = ldexp(-double(y.error.mantissa), y.error.exponent);
146  dp.upper_neg = nextafter(-center, -INFINITY) + wd;
147  dp.lower_pos = nextafter(center, -INFINITY) + wd;
148 }

181 {
182  MP_type zvalue;
183  sizetype zerror;
184  int local_prec;
185  const auto &stack = actual_stack();
186  if (stack.prec_policy == 0)
187  local_prec = max({y.error.exponent, this->error.exponent,
188  stack.actual_prec});
189  else {
190  local_prec = max(this->vsize.exponent, y.vsize.exponent);
191  local_prec = max({y.error.exponent, this->error.exponent,
192  local_prec - 50 + stack.actual_prec});

196 
197  sizetype_add_wo_norm(zerror, this->error, y.error);
198  zerror = sizetype_add_power2(zerror, local_prec);
199 
200  return REAL(zvalue, zerror);

252 std::string swrite(const REAL & x, const int w, const float_form form)
253 {
254  if (!x.value) {
255  REAL y(x);
256  y.mp_make_mp();
257  return swrite(y, w, form);
258  }
259 
260  std::string result;
261  if (get_cached(result))
262  return result;
263 
264  int width = w;
265  if (width < 9)
266  width = 9;
267  char * erg;
268 
269  switch (form) {
270  case float_form::absolute: {
271  sizetype psize;
272  int p = -10 * (width - 8) / 3;
273  psize = sizetype_power2(p);
274  int s = MP_size(x.value);
275  int mantissa =
276  (int)((s - x.error.exponent - GUARD_BITS) * .30103);
277  if (sizetype_less(psize, x.error) ||
278  (s > p && mantissa + 8 < width)) {
279  iRRAM_DEBUG2(1, "insufficient precision %d*2^(%d) in "
280  "conversion with precision 2^(%d)\n",
281  x.error.mantissa, x.error.exponent, p);
282  iRRAM_REITERATE(p - x.error.exponent);
283  }
284 
285  if (s <= p) {
286  erg = (char *)calloc(width + 1, sizeof(char));
287  strcpy(erg, " 0 ");
288  for (int j = 4; j < width; j++)
289  erg[j] = ' ';
290  erg[width] = 0;
291  } else {
292  erg = MP_swrite(x.value, width);
293  }
294  break;
295  }
296  case float_form::relative: {
297  int p;
298  {
299  single_valued code;
300  // We want to have an error less than 0.65 ulp, where
301  // 0.51 ulp come from
302  // the conversion to decimal, so we need to know that
303  // the error is at most
304  // 0.15ulp.
305  // At the time, we take the size of x....
306  // It would be better to take the x.size, subtract
307  // x.error and approximately
308  // convert the result to decimal in order to get a
309  // correct approximation
310  // for a lower bound of x. Using this lower bound we can
311  // estimate what
312  // 0.15ulp means in binary.
313  p = size(x) - 2 - 10 * (width - 8) / 3;
314  // We have to prevent that "size" changes the
315  // multi-value-cache here!
316  // Otherwise, any iteration would not use the same
317  // amount of
318  // cache as the first run!
319  }
320  sizetype psize;
321  psize = sizetype_power2(p);
322  int s = MP_size(x.value);
323  int mantissa =
324  (int)((s - x.error.exponent - GUARD_BITS) * .30103);
325  if (sizetype_less(psize, x.error) ||
326  (s > p && mantissa + 8 < width)) {
327  iRRAM_DEBUG2(1, "insufficient precision %d*2^(%d) in "
328  "writing with precision 2^(%d)\n",
329  x.error.mantissa, x.error.exponent, p);
330  iRRAM_REITERATE(p - x.error.exponent);
331  }
332  erg = MP_swrite(x.value, width);
333  break;
334  }
335  case float_form::show: {
336  int mantissa = (int)(int((MP_size(x.value) - x.error.exponent -
337  GUARD_BITS)) *
338  .30103);
339  if (mantissa < 1) {
340  erg = (char *)calloc(width + 1, sizeof(char));
341  sprintf(erg, " .*E%05.0f",
342  1 + (x.error.exponent + GUARD_BITS) * .30103);
343  for (int j = 9; j < width; j++)
344  erg[j] = ' ';
345  } else {
346  if (mantissa > width - 8)
347  mantissa = width - 8;
348  char * erg2 = MP_swrite(x.value, mantissa + 8);
349  erg = (char *)calloc(width + 1, sizeof(char));
350  strncpy(erg, erg2, mantissa + 8);
351  for (int j = mantissa + 8; j < width; j++)
352  erg[j] = ' ';
353  erg[width] = 0;
354  free(erg2);
355  }
356  break;
357  }
358  }
359 
360  result = erg;
361  free(erg);
362  put_cached(result);
363  return result;
364 }

479 REAL REAL::mp_division(const REAL & y) const
480 {
481  MP_type zvalue;
482  sizetype zerror, h1, h2, h3;
483  int local_prec;
484  sizetype_half(h1, y.vsize);
485  if (sizetype_less(h1, y.error)) {
486  iRRAM_DEBUG2(1, "insufficient precision %d*2^(%d) in "
487  "denominator of size %d*2^(%d)\n",
488  y.error.mantissa, y.error.exponent,
489  y.vsize.mantissa, y.vsize.exponent);
490  iRRAM_REITERATE(0);
491  }
492  h1 = this->vsize * y.error;
493  h2 = y.vsize * this->error;
494  h1 += h2;
495  h3 = y.vsize;
496  sizetype_dec(h3);
497  sizetype_dec(h3, y.error);
498  h2 = h3 * y.vsize;
499  sizetype_div(zerror, h1, h2);
500  const auto &stack = actual_stack();
501  if (stack.prec_policy == 0)
502  local_prec = max(zerror.exponent, stack.actual_prec);
503  else
504  local_prec = max(zerror.exponent,
505  this->vsize.exponent - y.vsize.exponent - 50 +
506  stack.actual_prec);
507  MP_init(zvalue);
508  MP_mv_div(this->value, y.value, zvalue, local_prec);
509  zerror = sizetype_add_power2(zerror, local_prec);
510  return REAL(zvalue, zerror);
511 }

1227 int module(REAL (*f)(const REAL&),const REAL& x, int p){
1228 // Semantics: If m=module(f,x,p), then
1229 // |x-z| <=2^m implies |f(x)-f(z)| <= 2^p
1230 
1231 // If we are able to approximate f(x) by a DYADIC number d with an error of <=2^{p-1},
1232 // then for any z in the argument interval of x, f(z) must differ by at most
1233 // 2^{p-1} from d, hence |f(x)-f(z)|<=2^p
1234 
1235  int result;
1236  if (get_cached(result))
1237  return result;
1238 
1239  DYADIC d;
1240  REAL x_copy=x;
1241 
1242 
1243  sizetype argerror,testerror;
1244 
1245  x_copy.geterror(argerror);
1246  testerror = sizetype_power2(argerror.exponent);
1247  x_copy.adderror(testerror);
1248  {
1249  single_valued code;
1250  d=approx(f(x_copy),p-1);
1251  }
1252 // At this line, we are sure that x_copy (and so also x) is precise enough to allow
1253 // the computation of f(x), even with a slightly increased error of the argument.
1254 
1255 // We now try to find the smallest p_arg such that the evaluation of f(x+- 2^p_arg)
1256 // is possible up to an error of at most 2^{p-1}
1257 
1258 // To do this, we start with p_arg=p.
1259 // If this is successfull, we increase the value of p_arg until the first failure
1260 // It it is not, then we decrease until the first success...
1261  int direction=0,p_arg=p;
1262  bool try_it=true;
1263 
1264  while (try_it) {
1265  testerror = sizetype_power2(p_arg);
1266  x_copy.seterror(argerror);
1267  x_copy.adderror(testerror);
1268  bool fail = false;
1269  if ( iRRAM_unlikely(state->debug > 0 ) ) {
1270  sizetype x_error;
1271  x_copy.geterror(x_error);
1272  iRRAM_DEBUG2(1,"Testing module: 1*2^%d + %d*2^%d\n",p_arg,argerror.mantissa,argerror.exponent);
1273  iRRAM_DEBUG2(1,"argument error: %d*2^%d\n",x_error.mantissa,x_error.exponent);
1274  }
1275  try {
1276  single_valued code;
1277  REAL z=f(x_copy);
1278  if ( iRRAM_unlikely(state->debug > 0 ) ) {
1279  sizetype z_error;
1280  z.geterror(z_error);
1281  iRRAM_DEBUG2(1,"Module yields result %d*2^%d\n",z_error.mantissa,z_error.exponent);
1282  }
1283  d=approx(z,p-1);
1284  }
1285  catch ( Iteration it) { fail=true; }
1286  switch ( direction ) {
1287  case 0:;
1288  if ( fail ) direction=-1; else direction=1;
1289  p_arg+=direction;
1290  break;
1291  case 1:;
1292  if ( fail ) { try_it=false; p_arg -=direction; }
1293  else { p_arg += direction; };
1294  break;
1295  case -1:;
1296  if ( fail ) { p_arg +=direction; }
1297  else { try_it=false; };
1298  break;
1299  }
1300  }
1301  iRRAM_DEBUG2(1,"Modules resulting in p_arg=%d\n",p_arg);
1302 
1303  testerror = sizetype_power2(p_arg);
1304  argerror += testerror;
1305

1307  argerror.mantissa=argerror.mantissa>>1;
1308  argerror.exponent+=1;
1309  }
1310
