74 inline void sizetype_normalize( sizetype& e) {
75  if (iRRAM_unlikely(e.mantissa < min_mantissa)) {
76  e.mantissa <<= BIT_RANGE;
77  e.exponent -= BIT_RANGE;
78  }
79  if (iRRAM_unlikely( e.mantissa >= max_mantissa ) ) {
80  e.mantissa = ( e.mantissa>> DIFF_BITS ) + 1;
81  e.exponent += DIFF_BITS;
82  }
83  if (iRRAM_unlikely( e.exponent < MP_min ) ){
84  e.exponent = min_exponent;
85  }
86  if (iRRAM_unlikely( e.exponent >= MP_max ) )
87  {
88  iRRAM_DEBUG1(1,"exponent too big in sizetype_normalize ");
89  iRRAM_REITERATE(0);
90  }
91 }

222 inline sizetype sizetype_add_power2(sizetype x,
223  typename sizetype::exponent_t exp)
224 {
225  /* leave as much of the original mantissa intact as possible */
226  if (exp < x.exponent + GUARD_BITS) {
227  x.mantissa += 1U << max(0, exp - x.exponent);
228  /* x.exponent remains unchanged */
229  } else {
230  x.mantissa = scale(x.mantissa, exp-(GUARD_BITS-1)-x.exponent)+1
231  + (1U << (GUARD_BITS-1));
232  x.exponent = exp-(GUARD_BITS-1);
233  }
234  sizetype_normalize(x);
235  return x;
236 }

314 { unsigned long long lmantissa=

406 inline bool sizetype_less(const sizetype& y,const sizetype& z)
407 { unsigned int mantissa;
408  if (iRRAM_unlikely(y.mantissa==0)) return true;
409  if (iRRAM_unlikely(z.mantissa==0)) return false;
410  if (y.exponent>z.exponent)
411  {
412  mantissa=scale(z.mantissa,y.exponent-z.exponent);
413  return (mantissa>=y.mantissa);
414  } else {
415  mantissa=scale(y.mantissa,z.exponent-y.exponent);
416  return (mantissa<z.mantissa);
417  }
418 }

433 inline int sizetype_log2(const sizetype &x)
434 {
435  int r;
436  typename sizetype::mantissa_t v = x.mantissa;
437  if (!v)
438  return min_exponent;
439  r = CHAR_BIT * sizeof(v) - clz(v) - 1; /* floor(log2(v)) */
440  r += (v & (v-1)) != 0; /* ceil(log2(v)) */
441  r += x.exponent;
442  return r;
443 }

478  "sizetype_sqrt() only proven for MANTISSA_BITS <= DBL_MANT_DIG");
