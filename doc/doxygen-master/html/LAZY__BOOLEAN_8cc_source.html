60 std::size_t choose(std::initializer_list<LAZY_BOOLEAN> x)
61 {
62  using std::find_if;
63 
64  std::size_t result;
65  if (get_cached(result))
66  return result;
67 
68  auto is_true = [](const LAZY_BOOLEAN &p)
69  { return p.value == true; };
70  auto is_bot = [](const LAZY_BOOLEAN &p)
71  { return p.value == LAZY_BOOLEAN::BOTTOM; };
72 
73  auto true_pos = find_if(begin(x), end(x), is_true);
74  if (true_pos != end(x)) {
75  result = 1 + (true_pos - begin(x));
76  } else if (find_if(begin(x), end(x), is_bot) != end(x)) {
77  iRRAM_DEBUG1(1,"choose(init-list): lazy boolean value BOTTOM "
78  "leading to iteration\n");
79  iRRAM_REITERATE(0);
80  } else {
81  result = 0;
82  }
83 
84  put_cached(result);
85  return result;
86 }
