82 template <class DATA>
83 class cache : public internal::cache_type
84 {
85  std::vector<typename internal::get_type<DATA>::type> data;
86  unsigned int current = 0;
87  bool active = false;
88 
89  void activate()
90  {
91  data.clear();
92  active = true;
93  state->cache_active->id[state->max_active++] = this;
94  current = 0;
95  }
96 
97 public:
98  void put(const DATA & x)
99  {
100  if (iRRAM_unlikely(!active))
101  activate();
102  data.emplace_back(x);
103  current++;
104  }
105 
106  bool get(DATA & x) noexcept(noexcept(x=x))
107  {
108  if (current >= data.size())
109  return false;
110  x = data[current++];
111  return true;
112  }
113 
114  void modify(const DATA & x) noexcept(noexcept(const_cast<DATA &>(x)=x))
115  {
116  data[current - 1] = x;
117  }
118 
119  void rewind() noexcept { current = 0; }
120 
121  void clear()
122  {
123  data.clear();
124  active = false;
125  current = 0;
126  }
127 };
