89 template <class DATA>
90 class cache : public cache_type
91 {
92  std::vector<typename get_type<DATA>::type> data;
93  unsigned int current = 0;
94  bool active = false;
95 
96  void activate()
97  {
98  data.clear();
99  active = true;
100  state->cache_active->id[state->max_active++] = this;
101  current = 0;
102  }
103 
104 public:
105  void put(const DATA & x)
106  {
107  if (iRRAM_unlikely(!active))
108  activate();
109  data.emplace_back(x);
110  current++;
111  }
112 
113  bool get(DATA & x) noexcept(noexcept(x=x))
114  {
115  if (current >= data.size())
116  return false;
117  x = data[current++];
118  return true;
119  }
120 
121  void modify(const DATA & x) noexcept(noexcept(const_cast<DATA &>(x)=x))
122  {
123  data[current - 1] = x;
124  }
125 
126  void rewind() noexcept { current = 0; }
127 
128  void clear()
129  {
130  data.clear();
131  active = false;
132  current = 0;
133  }
134 };
