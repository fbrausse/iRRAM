35 static void init_prec_array(int (&prec_array)[iRRAM_prec_steps],
36  int starting_prec, int _prec_inc,
37  double prec_factor, int debug)
38 {
39  prec_array[0] = 2100000000;
40  prec_array[1] = starting_prec;
41  double factor = std::sqrt(std::sqrt(prec_factor));
42  int prec_inc = _prec_inc;
43  if (debug)
44  std::cerr << "Basic precision bounds: "
45  << "double[1]";
46  for (int i = 2; i < iRRAM_prec_steps; i++) {
47  prec_array[i] = starting_prec + prec_inc;
48  prec_inc = int(prec_inc * factor) + _prec_inc;
49  if (prec_array[i] >= prec_array[i - 1])
50  prec_array[i] = prec_array[i - 1];
51  else if (debug && ((i % 5 == 0) || (i < 10)))
52  std::cerr << " " << prec_array[i] << "[" << i << "]";
53  }
54  if (debug)
55  std::cerr << "\n";
56 }

184 void internal::run::loop_fini(int p_end)
185 {
186  ITERATION_DATA &actual_stack = st.ACTUAL_STACK;
187  assert(st.highlevel == (actual_stack.prec_step > iRRAM_DEFAULT_PREC_START));
188 
189  int prec_skip = 0;
190  do {
191  prec_skip++;
192  code.inc_step(4);
193  } while ((actual_stack.actual_prec > p_end) &&
194  (prec_skip != st.prec_skip));
195 
196  assert(actual_stack.inlimit == 0);
197 
198  if (iRRAM_unlikely(st.debug > 0)) {
199  show_statistics();
200  if (st.max_prec <= actual_stack.prec_step)
201  st.max_prec = actual_stack.prec_step;
202  cerr << "increasing precision bound to "
203  << actual_stack.actual_prec << "["
204  << actual_stack.prec_step << "]\n";
205  }
206 }

274 extern "C" int iRRAM_parse_args(struct iRRAM_init_options *opts, int *argc, char **argv)
275 {
276  auto debug_enabled = [&opts](int level){ return opts->debug >= level; };
277 
278  for (int i = 1; i < *argc; i += 1) {
279  if (!strcmp(argv[i], "-d")) {
280  opts->debug = 1;
281  iRRAM_DEBUG2(1, "%s\n", "Debugging Mode");
282  } else
283  if (!strncmp(argv[i], "--debug=", 8)) {
284  opts->debug = atoi(&(argv[i][8]));
285  iRRAM_DEBUG2(1, "Debugging Level %d\n", opts->debug);
286  } else
287  if (!strncmp(argv[i], "--prec_init=", 12)) {
288  opts->starting_prec = atoi(&(argv[i][12]));
289  iRRAM_DEBUG2(1, "Initialising precision to 2^(%d)\n",
290  opts->starting_prec);
291  } else
292  if (!strncmp(argv[i], "--prec_inc=", 11)) {
293  int hi;
294  hi = atoi(&(argv[i][11]));
295  if (hi > 0)
296  opts->prec_inc = -hi;
297  iRRAM_DEBUG2(
298  1,
299  "Initialising precision increment to %d bits\n",
300  -opts->prec_inc);
301  } else
302  if (!strncmp(argv[i], "--prec_factor=", 14)) {
303  double hd;
304  hd = atof(&(argv[i][14]));
305  if (hd > 1.0)
306  opts->prec_factor = hd;
307  iRRAM_DEBUG2(1, "Initialising precision factor to %f\n",
308  opts->prec_factor);
309  } else
310  if (!strncmp(argv[i], "--prec_skip=", 12)) {
311  int hi;
312  hi = atoi(&(argv[i][12]));
313  if (hi > 0)
314  opts->prec_skip = hi;
315  iRRAM_DEBUG2(1, "Changed heuristic for precision "
316  "changes to skip at most %d steps\n",
317  opts->prec_skip);
318  } else
319  if (!strncmp(argv[i], "--prec_start=", 13)) {
320  int hi;
321  hi = atoi(&(argv[i][13]));
322  if (hi > 0)
323  opts->prec_start = hi;
324  iRRAM_DEBUG2(1,
325  "Changed inital precision step to %d \n",
326  opts->prec_start);
327  } else
328  if (!strcmp(argv[i], "-h") || !strcmp(argv[i], "--help")) {
329  fprintf(stderr,
330 "Runtime parameters for the iRRAM library:\n"
331 "--prec_init=n [%4d] starting precision\n"
332 "--prec_inc=n [%4d] basic increment for precision changes\n"
333 "--prec_factor=x [%4g] basic factor for precision changes\n"
334 "--prec_skip=n [%4d] bound for precision increments skipped by heuristic\n"
335 "--prec_start=n [%4d] initial precision level\n"
336 "--debug=n [%4d] level of limits up to which debugging should happen\n"
337 "-d debug mode, with level 1\n"
338 "-h / --help this help message\n",
339  opts->starting_prec,
340  opts->prec_inc,
341  opts->prec_factor,
342  opts->prec_skip,
343  opts->prec_start,
344  opts->debug);
345  } else
346  continue;
347  /* argument handled, remove from list */
348  for (int j = i+1; j < *argc; j++)
349  argv[j-1] = argv[j];
350  argv[--*argc] = NULL;
351  --i;
352  }
353  return iRRAM_success;
354 }
