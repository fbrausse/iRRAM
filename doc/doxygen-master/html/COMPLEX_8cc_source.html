102 static COMPLEX c_sqrt_approx(int p, int* choice, const COMPLEX& z)
103 {
104  REAL a,b,c,d,r;
105  COMPLEX y;
106  r= abs (z);
107  a= real(z);
108  b= imag(z);
109  c= sqrt( (r+a) / 2 );
110  d= sqrt( (r-a) / 2 );
111  if ( *choice == 0 ) {
112  if ( !bound(r, 2*p) ) {
113  if ( ! bound(b,p-10) ) {
114  if ( b > REAL(0) ) *choice=1;
115  else *choice=2;
116  } else {
117  if ( a > REAL(0) ) *choice=3;
118  else *choice=4;
119  }
120  }
121  }
122 
123  switch ( *choice ) {
124  case 0: y = COMPLEX ( 0 , 0 ); break;
125  case 1: y = COMPLEX ( c , d ); break;
126  case 2: y = COMPLEX ( c , -d ); break;
127  case 3: y = COMPLEX ( c , b/2/c ); break;
128  case 4: y = COMPLEX ( b/2/d , d ); break;
129  }
130  return y;
131 }

165 REAL arg(const COMPLEX& z){
166  REAL z_abs=abs(z);
167  COMPLEX z_norm= z/COMPLEX(z_abs);
168  REAL border=0.5;
169  REAL phi;
170  switch (choose(real(z_norm)>border,
171  imag(z_norm)>border,
172  imag(z_norm)<-border,
173  real(z_norm)<-border)
174  ){
175  case 1:phi= asin(imag(z_norm)); break;;
176  case 2:phi= acos(real(z_norm)); break;;
177  case 3:phi= -acos(real(z_norm)); break;;
178  case 4:phi= pi()-asin(imag(z_norm)); break;;
179  }
180  if (positive(phi-pi(),-10)) phi=phi-2*pi();
181  return phi;
182 }

212 COMPLEX sin (const COMPLEX& z)
213 {
214  REAL scalar, iz_re, iz_im, re1, re2, im1, im2;
215 
216  iz_re = -imag(z);
217  iz_im = real(z);
218  scalar = exp(iz_re);
219  re1 = scalar * cos(iz_im);
220  im1 = scalar * sin(iz_im);
221  scalar = exp(-iz_re);
222  re2 = scalar * cos(-iz_im);
223  im2 = scalar * sin(-iz_im);
224  re1 = re1 - re2;
225  im1 = im1 - im2;
226

228 }
