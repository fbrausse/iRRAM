46 void mpfr_ext_test (mpfr_ptr y, mpfr_srcptr u, int p, mp_rnd_t rnd_mode)
47  {
48  mpfr_t r;
49  mpfr_init(r);
50  int err=50;
51  if (p< 0) err= p;
52  do {
53  err=err-10;
54  mpfr_wrapper(compute_test,r,u,err);
55 // printf(" %d %d %d\n",mpfr_get_exp(r),err,mpfr_get_exp(r)+err);
56  } while (! mpfr_can_round(r,mpfr_get_exp(r)+err ,__gmp_default_rounding_mode, rnd_mode, p ));
57 
58  mpfr_set_prec(y,p);
59  mpfr_set(y,r,rnd_mode);
60  mpfr_clear(r);
61  }
