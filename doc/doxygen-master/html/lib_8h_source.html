94  template <typename F,typename... Args>
95  ret_void_t<F,Args...> exec(F f, const Args &... args)
96  {
97  ITERATION_DATA &actual_stack = st.ACTUAL_STACK;
98  while (true) {
99  loop_init();
100 
101  int p_end = 0;
102  try {
103  f(args...);
104  if (iRRAM_likely(!st.infinite))
105  break;
106  } catch (const Iteration &it) {
107  p_end = actual_stack.actual_prec + it.prec_diff;
108  } catch (const iRRAM_Numerical_Exception &exc) {
109  /* not handled by run::exec */
110  cerr << "iRRAM exception: "
111  << iRRAM_error_msg[exc.type] << "\n";
112  throw;
113  }
114 
115  loop_fini(p_end);
116  }
117  }
118 
119  template <typename F,typename... Args>
120  ret_value_t<F,Args...> exec(F f, const Args &... args)
121  {
122  ret_value_t<F,Args...> r;
123  exec([f,&r](const Args &... args){ r = f(args...); }, args...);
124  return r;
125  }
