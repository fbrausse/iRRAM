58 template <typename T>
59 static inline unsigned clz(T x)
60 {
61  unsigned r = x != 0;
62  for (unsigned m = (sizeof(x) * CHAR_BIT) >> 1; m > 0; m >>= 1)
63  if (x >= (T)1 << m) {
64  r += m;
65  x >>= m;
66  }
67  return sizeof(x) * CHAR_BIT - r;
68 }
69 #if defined(__GNUC__) || defined(__clang__)
70 # define CLZ_BODY(x,clz) { return x ? clz(x) : sizeof(x) * CHAR_BIT; }
71 template <> inline unsigned clz(unsigned x) CLZ_BODY(x,__builtin_clz)
72 template <> inline unsigned clz(unsigned long x) CLZ_BODY(x,__builtin_clzl)
73 template <> inline unsigned clz(unsigned long long x) CLZ_BODY(x,__builtin_clzll)
74 # undef CLZ_BODY
75 #endif
