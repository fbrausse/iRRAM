32 void sparse_set(SPARSEREALMATRIX& m,unsigned int i,unsigned int j,
33  const REAL& x){
34  SPM_ELEMENT* ptr_r= m.rowvector[i];
35  if ( m.hotspot != NULL && m.hotspot->rowindex==i && m.hotspot->colindex<=j)
36  ptr_r=m.hotspot;
37  if ( ptr_r!=NULL ) {
38  if (ptr_r->colindex==j) {ptr_r->value=x;return;}
39  while ( (ptr_r->nextcol != NULL)&& (ptr_r->nextcol->colindex <= j) ) {
40  ptr_r=ptr_r->nextcol;
41  if (ptr_r->colindex==j) {ptr_r->value=x;return;}
42  }
43  }
44  SPM_ELEMENT* ptr= new SPM_ELEMENT;
45  m.filled+=1;
46  ptr->nextcol=NULL;
47  ptr->nextrow=NULL;
48  ptr->colindex=j;
49  ptr->rowindex=i;
50  ptr->value=x;
51 
52  if ( m.rowvector[i]!=NULL && m.rowvector[i]->colindex < j) {
53  ptr->nextcol=ptr_r->nextcol;
54  ptr_r->nextcol=ptr;
55  } else {
56  ptr->nextcol=m.rowvector[i];
57  m.rowvector[i] = ptr;
58  }
59  SPM_ELEMENT* ptr_c= m.colvector[j];
60  if ( ptr_c!=NULL && m.colvector[j]->rowindex < i ) {
61  while (ptr_c->nextrow != NULL && ptr_c->nextrow->rowindex < i)
62  ptr_c=ptr_c->nextrow;
63  ptr->nextrow=ptr_c->nextrow;
64  ptr_c->nextrow=ptr;
65  } else {
66  ptr->nextrow=m.colvector[j];
67  m.colvector[j] = ptr;
68  }
69  m.hotspot=ptr;
70 }

284 SPARSEREALMATRIX operator + (const SPARSEREALMATRIX& x, const SPARSEREALMATRIX& y) {
285  SPARSEREALMATRIX sum(x.maxrow,x.maxcolumn);
286 
287  SPM_ELEMENT* ptr_x;
288  SPM_ELEMENT* ptr_y;
289 
290  if ( x.maxrow != y.maxrow || x.maxcolumn != y.maxcolumn) {
291  fprintf(stderr,"Error in adding real matrices of different sizes \n");
292  exit(1);
293  }
294  for (unsigned int i=0;i<x.maxrow;i++) {
295  ptr_y=y.rowvector[i];
296  ptr_x=x.rowvector[i];
297  while (ptr_x != NULL || ptr_y != NULL) {
298  if ( ptr_y == NULL ) {
299  sparse_set(sum,i,ptr_x->colindex,ptr_x->value);
300  ptr_x=ptr_x->nextcol;
301  } else if ( ptr_x == NULL ) {
302  sparse_set(sum,i,ptr_y->colindex,ptr_y->value);
303  ptr_y=ptr_y->nextcol;
304  } else if ( ptr_x->colindex < ptr_y->colindex) {
305  sparse_set(sum,i,ptr_x->colindex,ptr_x->value);
306  ptr_x=ptr_x->nextcol;
307  } else if ( ptr_x->colindex > ptr_y->colindex) {
308  sparse_set(sum,i,ptr_y->colindex,ptr_y->value);
309  ptr_y=ptr_y->nextcol;
310  } else {
311  sparse_set(sum,i,ptr_y->colindex,ptr_x->value+ptr_y->value);
312  ptr_x=ptr_x->nextcol;
313  ptr_y=ptr_y->nextcol;
314  }
315  }
316  }
317  return sum;

460 SPARSEREALMATRIX solve (SPARSEREALMATRIX& lside, SPARSEREALMATRIX& rside,int use_pivot) {
461  single_valued code;
462  SPARSEREALMATRIX result(rside.maxrow,rside.maxcolumn);
463  SPM_ELEMENT* ptr;
464  REAL faktor;
465  SPM_ELEMENT* ptr2;
466  REAL faktor2;
467  SPM_ELEMENT* ptr3;
468  int testsize,pivotsize,pivot,pivotsave;
469  std::vector<int> perm(lside.maxrow), backperm(lside.maxrow);
470 
471 
472  if ( lside.maxrow != rside.maxrow) {
473  fprintf(stderr,"Error, real matrices of different sizes \n");
474  exit(1);
475  }
476  for (unsigned int i=0; i<lside.maxrow;i++) perm[i]=i;
477  for (unsigned int i=0; i<lside.maxrow;i++) backperm[i]=i;
478 
479  for (unsigned int i=0; i < lside.maxrow; i++) {
480  pivot=i;
481  if (use_pivot) {
482  pivotsize=upperbound(lside(perm[i],i));
483  for (unsigned int j=i+1; j < lside.maxcolumn; j++) {
484  testsize=upperbound(lside(perm[j],i));
485  if (testsize>pivotsize) {
486  pivot=j;
487  pivotsize=testsize;
488  }
489  else if ( testsize < pivotsize && use_pivot == 1) break ;
490  }
491  pivotsave=perm[pivot];
492  perm[pivot]=perm[i];
493  perm[i]=pivotsave;
494  }
495  backperm[perm[i]]=i;
496  backperm[perm[pivot]]=pivot;
497  ptr=lside.colvector[i];
498  while (ptr != NULL ) {
499  unsigned int j=backperm[ptr->rowindex];
500  if ( j > i) {
501  faktor=ptr->value/lside(perm[i],i);
502  ptr2=rside.rowvector[perm[i]];
503  while (ptr2 != NULL ) {
504  unsigned int k=ptr2->colindex;
505  sparse_set(rside,perm[j],k,
506  rside(perm[j],k)
507  -faktor*ptr2->value);
508  ptr2=ptr2->nextcol;
509  }
510  ptr2=lside.rowvector[perm[i]];
511  while (ptr2 != NULL ) {
512  unsigned int l=ptr2->colindex;
513  if ( l>=i )
514  sparse_set(lside,perm[j],l,
515  lside(perm[j],l)
516  -faktor*ptr2->value);
517  ptr2=ptr2->nextcol;
518  }
519  }
520  ptr=ptr->nextrow;
521  }
522  }
523 
524  for (int i=rside.maxrow-1; i >= 0; i--) {
525  faktor=1/lside(perm[i],i);
526 
527  for (int k=rside.maxcolumn-1; k>=0; k--) {
528  ptr2=sparse_get_ptr(rside,perm[i],k);
529  if (ptr2 != NULL ) {
530  sparse_set(result,i,k,ptr2->value*faktor);
531  faktor2=result(i,k);
532 
533  ptr3=lside.colvector[i];
534  while (ptr3 != NULL) {
535  int j= backperm[ptr3->rowindex];
536  if ( j < i ) sparse_set(rside,perm[j],k,
537  rside(perm[j],k)
538  -ptr3->value*faktor2);
539  ptr3=ptr3->nextrow;
540  } }
541  } }
542  return result;
543 }
