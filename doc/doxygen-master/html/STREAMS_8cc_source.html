119 template <class PARAM>
120 static orstream & iRRAM_out(orstream * s, const PARAM & x)
121 {
122  if (actual_stack().inlimit > 0 && s->_respect_iteration) {
123  iRRAM_DEBUG1(2, "illegal output in continuous section!\n");
124  return *s;
125  }
126  if (actual_stack().inlimit == 0 && s->_respect_iteration) {
127  single_valued code;
128  if (++state->requests > state->outputs) {
129  *s->target << x;
130  state->outputs++;
131  }
132  } else {
133  *s->target << x;
134  }
135  return *s;
136 }
