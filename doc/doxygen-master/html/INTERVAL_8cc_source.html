53 static INTERVAL mul_pos_x(const INTERVAL &x, const INTERVAL &y, int y_sgn)
54 {
55  if (y_sgn > 0)
56  return INTERVAL(x.low * y.low, x.upp * y.upp, true);
57  if (y_sgn < 0)
58  return INTERVAL(x.upp * y.low, x.low * y.upp, true);
59  return INTERVAL(minimum(x.upp * y.low, x.low * y.low),
60  maximum(x.upp * y.upp, x.low * y.upp), true);
61 }

63 INTERVAL operator * (const INTERVAL & x, const INTERVAL & y)
64 {
65  int x_sgn, y_sgn;
66  {
67  single_valued code;
68  x_sgn = (x.low > 0).value == true ? +1
69  : (x.upp < 0).value == true ? -1 : 0;
70  y_sgn = (y.low > 0).value == true ? +1
71  : (y.upp < 0).value == true ? -1 : 0;
72  }
73  if (x_sgn > 0)
74  return mul_pos_x(x, y, y_sgn);
75  if (y_sgn > 0)
76  return mul_pos_x(y, x, x_sgn);
77  if (x_sgn < 0)
78  return -mul_pos_x(-x, y, y_sgn);
79  if (y_sgn < 0)
80  return -mul_pos_x(-y, x, x_sgn);
81  REAL aa= x.low*y.low;
82  REAL ab= x.upp*y.low;
83  REAL ba= x.low*y.upp;
84  REAL bb= x.upp*y.upp ;
85  REAL mina=minimum(aa,ab);
86  REAL minb=minimum(ba,bb);
87  REAL maxa=maximum(aa,ab);
88  REAL maxb=maximum(bb,ba);
89 
90  return INTERVAL(minimum(mina,minb),maximum(maxa,maxb), true);
91 
92 }

136 INTERVAL power(const INTERVAL& x,int n){
137  if (n==0) return INTERVAL(1);
138  if (n==1) return x;
139  if (n==2) return square(x);
140  if (n<0) return power(INTERVAL(1)/x,-n);
141  if (n%2==1)return INTERVAL(power(x.low,n),power(x.upp,n), true);
142  REAL l=power(x.low,n-1);
143  REAL u=power(x.upp,n-1);
144  return INTERVAL(l*maximum(x.low,REAL(0)) + u*minimum(x.upp,REAL(0)),
145  maximum(abs(l*x.low),abs(u*x.upp)), true);
146 }

158 INTERVAL sin(const INTERVAL& x){
159 // first, using lazy booleans, we do a multivalued test
160 // whether the interval width is definitely at least 2*pi.
161 
162  int test=choose( x.upp-x.low >= 2*pi(),
163  x.upp-x.low < 3*pi());
164 
165 // if the width is at least 2*pi, return the closed interval [-1,1]
166 
167  if ( test == 1 ) return INTERVAL(REAL(-1), REAL(1));
168 
169 // now we have a width of at most 3*pi,
170 // and we do a range reduction
171 
172  REAL inf_reduced=modulo(x.low,2*pi());
173  REAL sup_reduced=x.upp- (x.low-inf_reduced);
174 
175 // now we have -2*pi <= inf_reduced<= sup_reduced <= 5*pi
176 // with a difference between the values of at most 3*pi
177 
178 // we could estimate which odd multiples of pi/2 lie between the
179 // borders of the reduced interval, but for simplicity here I try all
180 // 7 possibilities (it was easier using copy and paste...)
181 
182 // the following values are either
183 // inf_reduced, or
184 // sup_reduced, or
185 // the multiple of pi/2, in case it lies between the two values
186 REAL p1= maximum(inf_reduced,minimum(sup_reduced,-3*pi()/2));
187 REAL p2= maximum(inf_reduced,minimum(sup_reduced, -pi()/2));
188 REAL p3= maximum(inf_reduced,minimum(sup_reduced, pi()/2));
189 REAL p4= maximum(inf_reduced,minimum(sup_reduced, 3*pi()/2));
190 REAL p5= maximum(inf_reduced,minimum(sup_reduced, 5*pi()/2));
191 REAL p6= maximum(inf_reduced,minimum(sup_reduced, 7*pi()/2));
192 REAL p7= maximum(inf_reduced,minimum(sup_reduced, 9*pi()/2));
193 REAL s1=sin(p1);
194 REAL s2=sin(p2);
195 REAL s3=sin(p3);
196 REAL s4=sin(p4);
197 REAL s5=sin(p5);
198 REAL s6=sin(p6);
199 REAL s7=sin(p7);
200 
201 REAL inf_min=minimum(sin(inf_reduced),sin(sup_reduced));
202 inf_min=minimum(inf_min,s1);
203 inf_min=minimum(inf_min,s2);
204 inf_min=minimum(inf_min,s3);
205 inf_min=minimum(inf_min,s4);
206 inf_min=minimum(inf_min,s5);
207 inf_min=minimum(inf_min,s6);
208 inf_min=minimum(inf_min,s7);
209 
210 REAL sup_min=maximum(sin(inf_reduced),sin(sup_reduced));
211 sup_min=maximum(sup_min,s1);
212 sup_min=maximum(sup_min,s2);
213 sup_min=maximum(sup_min,s3);
214 sup_min=maximum(sup_min,s4);
215 sup_min=maximum(sup_min,s5);
216 sup_min=maximum(sup_min,s6);
217 sup_min=maximum(sup_min,s7);
218 return INTERVAL(inf_min,sup_min);
219 }

287 // the interval x, then the monotonicity of tan() is sufficient
