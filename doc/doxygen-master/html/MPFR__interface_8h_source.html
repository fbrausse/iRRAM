293 inline mpfr_ptr ext_mpfr_init(struct iRRAM_ext_mpfr_cache_t *cache)
294 {
295  mpfr_ptr z;
296  if (cache->free_var_count > 0) {
297  cache->free_var_count -= 1;
298  z = cache->free_vars[cache->free_var_count];
299  } else {
300  z = (mpfr_ptr)malloc(sizeof(mpfr_t));
301  mpfr_init(z);
302  cache->total_alloc_var_count++;
303  }
304  cache->ext_mpfr_var_count += 1;
305 
306  /* fprintf(stderr,"create %x\n",z); */
307 
308  return z;
309 }

313 inline void ext_mpfr_free(struct iRRAM_ext_mpfr_cache_t *cache, mpfr_ptr z)
314 {
315  /* fprintf(stderr,"delete %x\n",z); */
316  if (cache->free_var_count < iRRAM_EXT_MPFR_CACHE_SIZE) {
317  cache->free_vars[cache->free_var_count] = z;
318  cache->free_var_count += 1;
319  } else {
320  mpfr_clear(z);
321  free(z);
322  cache->total_freed_var_count++;
323  }
324  cache->ext_mpfr_var_count -= 1;
325 }
