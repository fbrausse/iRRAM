379 inline REAL maximum (const REAL& x, const REAL& y)
380 {
381  if (iRRAM_unlikely(x.value || y.value))
382  return mp_maximum(x, y);
383  REAL::double_pair dp;
384  dp.lower_pos = fmax(x.dp.lower_pos, y.dp.lower_pos);
385  dp.upper_neg = fmin(x.dp.upper_neg, y.dp.upper_neg);
386  return REAL(dp);
387 }
388 
389 inline REAL minimum (const REAL& x, const REAL& y)
390 {
391  if (iRRAM_unlikely(x.value || y.value))
392  return mp_minimum(x, y);
393  REAL::double_pair dp;
394  dp.lower_pos = fmin(x.dp.lower_pos, y.dp.lower_pos);
395  dp.upper_neg = fmax(x.dp.upper_neg, y.dp.upper_neg);
396  return REAL(dp);
397 }

688 inline REAL operator*(const REAL & x, const REAL & y)
689 {
690  if (iRRAM_unlikely(x.value || y.value))
691  return x.mp_conv().mp_multiplication(y.mp_conv());
692  REAL::double_pair z;
693  if (x.dp.lower_pos >= 0 && y.dp.lower_pos >= 0) {
694  z.lower_pos = x.dp.lower_pos * y.dp.lower_pos;
695  z.upper_neg = (-x.dp.upper_neg) * y.dp.upper_neg;
696  } else if (x.dp.upper_neg >= 0 && y.dp.upper_neg >= 0) {
697  z.lower_pos = x.dp.upper_neg * y.dp.upper_neg;
698  z.upper_neg = (-x.dp.lower_pos) * y.dp.lower_pos;
699  } else if (x.dp.upper_neg >= 0 && y.dp.lower_pos >= 0) {
700  z.lower_pos = x.dp.lower_pos * (-y.dp.upper_neg);
701  z.upper_neg = x.dp.upper_neg * y.dp.lower_pos;
